## 💡 Sample Code for Building UI

```javascript
import * as AIInterp from './services/ai/aiInterpretationEngine';
import * as InsightGen from './services/ai/insightsGenerator';
import * as RecEngine from './services/ai/recommendationEngine';

// In your React component:
const [results, setResults] = useState(null);

const handleRunModel = async (inputs) => {
  // Step 1: Map and validate input (Phase 2)
  const mapped = InputMapping.mapUserInputToModel(inputs, 'forecasting');
  
  // Step 2: Run calculation (Phase 1)
  const calcResults = runForecastModel(mapped.mappedFields);
  
  // Step 3: Get interpretation (Phase 3)
  const interp = AIInterp.interpretForecastingResults(calcResults);
  
  // Step 4: Get insights (Phase 3)
  const insights = InsightGen.generateForecastInsights(calcResults);
  const prioritized = InsightGen.prioritizeInsights(insights);
  
  // Step 5: Get recommendations (Phase 3)
  const recs = RecEngine.generateForecastRecommendations(calcResults);
  const sorted = RecEngine.prioritizeRecommendations(recs);
  
  // Store for display
  setResults({
    interpretation: interp,
    insights: prioritized,
    recommendations: sorted
  });
};

// Display in UI:
<div>
  <InterpretationSection data={results.interpretation} />
  <InsightsSection data={results.insights} />
  <RecommendationsSection data={results.recommendations} />
</div>
```
